clear, clc
b = 1;
ul = 5;

step = ul/100;
f = @(x) (x/b^2)*exp((-x^2)/(2*(b^2)));
x = 0:step:ul-step;
c = zeros(length(x),1);
for i = 1:length(x)
    c(i) = f(x(i));
end
plot(x,c)
